const connect = Component => {
  const selectProps = Component.stateToProps;
  const extraProps = ({ store, selectors }) =>
    typeof selectProps === "function"
      ? selectProps(store, selectors)
      : { store, selectors };
  return React.forwardRef((props, ref) => {
    return (

          <Component
            {...props}
            {...extraProps(data)}
            ref={ref}
